## 实现添加动态路由

我在写vue3_admin的template时候我的三级路由不显示，问题是当时我二级路由的component指向的是一个文件地址，文件地址是一个具体的页面。

错误例子如下：
```
{
      path: '/首页',// 第一层路由
      name: '首页',
       meta: {
            name: '首页'
          },
      component: Layout,
      children: [  // 第二层路由
        {
          path: '/二层路由',
          name: '二层路由',
          component: import('@/views/二级路由/index.vue'),      // 指定的正常页面，所以导致三级路由页面不显示
          children: [ // 第三层路由
            {
              path: '三层路由',
              name: '三层路由',
              component: import('@/views/三级路由/index.vue'),
            },
          ]
        }
      ]
    }

```

所以需要改成以下代码
```
{
    path: '/首页',// 第一层路由
    name: '首页',
    meta: {
        name: '首页'
        },
    component: Layout,
    children: [  // 第二层路由
    {
        path: '/二层路由',
        name: '二层路由',
        component: subView,      // 这里要引用页面代码看下面代码
        children: [ // 第三层路由
        {
            path: '三层路由',
            name: '三层路由',
            component: import('@/views/三级路由/index.vue'),
        },
        ]
    }
    ]
}


// subView.vue  这里也要嵌套一个router-view 这样就可以正常显示了
<template>
    <router-view></router-view>
</template>
```
